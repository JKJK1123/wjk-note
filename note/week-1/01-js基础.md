##js基础

		<!-- 弹出一句话内容为一个 字符窜 类型 -->
		alert('欢迎来到不凡学院')

	外链式
	</script>

###confirm
	if...else...  是js编程中的关键词 逻辑判断语句  意思二选一
	confirm('你好吗?')将会返回一个 肯定(true)/否定(false) 的类型
		<script>
			if(confirm('你好吗?')){
				alert('好的!');
			}else{
				alert('真不好!');
			}
		</script>

###console
	一行代码结束后 需要以分号结尾
	调试js代码的生活需要大量用到console.log
		console.log('今天起真不错');

###prompt
		<script>
			prompt('请输入您的年龄')
		</script>

###write
		<script>
			document.write('给页面输出内容')
		</script>
