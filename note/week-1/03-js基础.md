###switch
	var fruit  = prompt('请输入水果名称!');
		//在有限的类别判断时候,可以使用,代码逻辑更清晰
		switch (fruit) {
			case '苹果':
				alert('苹果很好吃!');
				break;
			case '梨':
				alert('梨很脆!');
				break;
			case '香蕉':
				alert('香蕉很甜!');
				break;
			default:
				alert('我也没吃过!');
				break;
