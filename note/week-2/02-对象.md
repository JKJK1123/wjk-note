	 3. 创建对象的第三种方式
	 	在js中所有的对象都成为函数,所有的函数也可以称为'对象'

	  function Human (){
	  	this.name = '张三';
	  	this.age = 20;
	  	this.say = function(){
	  		alert(this.name+'的年龄为:'+this.age);
	  	}
	  }
	  	var zs = new Human();	//zs为通过Human创建的一个实例对象

	  	zs.say();

###this

		this在调用之前 是不知道指向谁的
		重点:

		1. 构造函数对象this指向new出来的实例对象

				var name = '李四';

					this.name = '张三';
					this.age = 20;

						alert(this.name+'的年龄为:'+this.age);
